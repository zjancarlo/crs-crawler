	function searchcourses() {
		var coursenumber = $.trim($('#txt_coursenumber').val());
		var alphanumeric_pattern = /^[a-zA-Z0-9\s\.\-]+$/;

		if(coursenumber != '') {

				alert('Course number must be alphanumeric.');

			}
		}
